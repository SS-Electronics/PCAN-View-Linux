65static void fmt_double(char *buf, size_t sz, double v)
66{
67 snprintf(buf, sz, "%.4f", v);
68 /* Trim trailing zeros and a dangling decimal point. */
69 char *dot = strchr(buf, '.');
70 if (!dot)
71 return;
72 char *end = buf + strlen(buf) - 1;
73 while (end > dot && *end == '0')
74 *end-- = '\0';
75 if (end == dot)
76 *end = '\0';
77}

86static void sig_table_rebuild(void)
87{
88 if (!g_gui.sig_store)
89 return;
90
91 gtk_list_store_clear(g_gui.sig_store);
92 if (!s_db)
93 return;
94
95 for (size_t mi = 0; mi < s_db->message_count; mi++) {
96 const dbc_message_t *m = &s_db->messages[mi];
97 char id_buf[16];
98 gui_format_id(id_buf, sizeof(id_buf), m->id, m->is_extended);
99
100 for (uint16_t si = 0; si < m->signal_count; si++) {
101 const dbc_signal_t *s = &m->signals[si];
102 char range[80], minb[32], maxb[32];
103 fmt_double(minb, sizeof(minb), s->min);
104 fmt_double(maxb, sizeof(maxb), s->max);
105 snprintf(range, sizeof(range), "%s .. %s", minb, maxb);
106
107 GtkTreeIter it;
108 gtk_list_store_append(g_gui.sig_store, &it);
109 gtk_list_store_set(g_gui.sig_store, &it,
110 SCOL_MSG, m->name,
111 SCOL_ID, id_buf,
112 SCOL_SIGNAL, s->name,
113 SCOL_RAW, "—",
114 SCOL_VALUE, "—",
115 SCOL_UNIT, s->unit,
116 SCOL_RANGE, range,
117 SCOL_COUNT, 0u,
118 SCOL_FG, NULL,
119 SCOL_MSGID, m->id,
120 SCOL_EXT, (gboolean)m->is_extended,
121 SCOL_SIGIDX, (guint)si,
122 -1);
123 }
124 }
125}

152gboolean gui_signal_load_dbc(const char *path)
153{
154 if (!path || !*path)
155 return FALSE;
156
157 char err[256] = {0};
158 dbc_db_t *db = dbc_load_file(path, err, sizeof(err));
159 if (!db) {
160 gui_status_message("DBC load failed: %s", err);
161 return FALSE;
162 }
163
164 if (s_db)
165 dbc_free(s_db);
166 s_db = db;
167
168 sig_table_rebuild();
169 gui_plot_set_database(s_db);
170 sig_update_dbc_label();
171 gui_status_message("Loaded database %s (%zu messages, %zu signals).",
172 db->path, db->message_count, db->signal_count);
173 return TRUE;
174}

201void gui_signal_decode_message(const can_msg_t *msg)
202{
203 if (!s_db || !g_gui.sig_store || !msg || msg->is_error)
204 return;
205
206 const dbc_message_t *m = dbc_find_message(s_db, msg->id, msg->is_extended);
207 if (!m)
208 return;
209
210 GtkTreeIter it;
211 if (!gtk_tree_model_get_iter_first(GTK_TREE_MODEL(g_gui.sig_store), &it))
212 return;
213
214 do {
215 guint row_id = 0, sig_idx = 0, cnt = 0;
216 gboolean row_ext = FALSE;
217 gtk_tree_model_get(GTK_TREE_MODEL(g_gui.sig_store), &it,
218 SCOL_MSGID, &row_id,
219 SCOL_EXT, &row_ext,
220 SCOL_SIGIDX, &sig_idx,
221 SCOL_COUNT, &cnt,
222 -1);
223
224 if (row_id != m->id ||
225 row_ext != (gboolean)m->is_extended ||
226 sig_idx >= m->signal_count)
227 continue;
228
229 const dbc_signal_t *s = &m->signals[sig_idx];
230 int64_t sraw = 0;
231 uint64_t raw = dbc_extract_raw(msg->data, msg->dlc, s);
232 double phys = dbc_decode_physical(s, raw, &sraw);
233
234 /* Feed the time-graph viewer with this decoded sample. */
235 gui_plot_add_sample(m->id, m->is_extended, (int)sig_idx, phys);
236
237 char raw_buf[32], val_buf[40];
238 snprintf(raw_buf, sizeof(raw_buf), "%lld", (long long)sraw);
239 fmt_double(val_buf, sizeof(val_buf), phys);
240
241 gtk_list_store_set(g_gui.sig_store, &it,
242 SCOL_RAW, raw_buf,
243 SCOL_VALUE, val_buf,
244 SCOL_COUNT, cnt + 1,
245 SCOL_FG, msg->direction == CAN_DIR_TX ? "blue" : "black",
246 -1);
247 } while (gtk_tree_model_iter_next(GTK_TREE_MODEL(g_gui.sig_store), &it));
248}
