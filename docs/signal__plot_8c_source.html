121void gui_plot_set_database(const dbc_db_t *db)
122{
123 plot_free_series();
124
125 if (!db || db->signal_count == 0) {
126 if (s_plot.area)
127 gtk_widget_queue_draw(s_plot.area);
128 return;
129 }
130
131 s_plot.series = g_malloc0(sizeof(plot_series_t) * db->signal_count);
132 int n = 0;
133
134 for (size_t mi = 0; mi < db->message_count; mi++) {
135 const dbc_message_t *m = &db->messages[mi];
136 for (uint16_t si = 0; si < m->signal_count; si++) {
137 const dbc_signal_t *s = &m->signals[si];
138 plot_series_t *ps = &s_plot.series[n];

140 snprintf(ps->label, sizeof(ps->label), "%s.%s", m->name, s->name);
141 snprintf(ps->unit, sizeof(ps->unit), "%s", s->unit);
142 ps->id = m->id;
143 ps->ext = m->is_extended;
144 ps->sig_idx = si;
145 ps->dmin = s->min;
146 ps->dmax = s->max;
147 ps->enabled = FALSE;
148 ps->col[0] = PALETTE[n % PALETTE_N][0];
149 ps->col[1] = PALETTE[n % PALETTE_N][1];
150 ps->col[2] = PALETTE[n % PALETTE_N][2];
151
152 char swatch[2 * DBC_NAME_MAX + 16];
153 snprintf(swatch, sizeof(swatch), "%s", ps->label);

156 /* Colour the label text to match the trace. */
157 char css[96];
158 snprintf(css, sizeof(css),

185void gui_plot_add_sample(uint32_t id, int is_ext, int sig_idx, double value)
186{
187 if (s_plot.paused || !s_plot.series)
188 return;
189
190 for (int i = 0; i < s_plot.n_series; i++) {
191 plot_series_t *ps = &s_plot.series[i];
192 if (ps->id != id || ps->ext != (is_ext ? 1 : 0) || ps->sig_idx != sig_idx)
193 continue;
194
195 ps->t[ps->head] = plot_now();
196 ps->v[ps->head] = value;
197 ps->head = (ps->head + 1) % PLOT_RING;
198 if (ps->count < PLOT_RING)
199 ps->count++;
200 ps->last = value;
201 ps->has_last = TRUE;
202 s_plot.dirty = TRUE;
203 return;
204 }
205}

214static double norm_to_y(double norm, double y0, double y1)
215{
216 if (norm < 0.0) norm = 0.0;
217 if (norm > 1.0) norm = 1.0;
218 return y1 - norm * (y1 - y0);
219}

224static gboolean on_draw(GtkWidget *w, cairo_t *cr, gpointer data)
225{
226 (void)data;
227 GtkAllocation a;
228 gtk_widget_get_allocation(w, &a);
229 const double W = a.width, H = a.height;
230
231 /* Background. */
232 cairo_set_source_rgb(cr, 0.12, 0.12, 0.14);
233 cairo_rectangle(cr, 0, 0, W, H);
234 cairo_fill(cr);
235
236 const double left = 50, right = 12, top = 10, bottom = 24;
237 const double x0 = left, x1 = W - right, y0 = top, y1 = H - bottom;
238 if (x1 <= x0 + 10 || y1 <= y0 + 10)
239 return FALSE;
240
241 const double now = plot_now();
242 const double winw = s_plot.window_sec;
243 const double tmin = now - winw;
244
245 /* Grid + Y axis labels (normalised 0..100 %). */
246 cairo_set_line_width(cr, 1.0);
247 cairo_select_font_face(cr, "monospace",
248 CAIRO_FONT_SLANT_NORMAL, CAIRO_FONT_WEIGHT_NORMAL);
249 cairo_set_font_size(cr, 10);
250 for (int i = 0; i <= 10; i++) {
251 double yy = y0 + (y1 - y0) * i / 10.0;
252 cairo_set_source_rgb(cr, 0.22, 0.22, 0.25);
253 cairo_move_to(cr, x0, yy);
254 cairo_line_to(cr, x1, yy);
255 cairo_stroke(cr);
256 char lbl[8];
257 snprintf(lbl, sizeof(lbl), "%d%%", 100 - i * 10);
258 cairo_set_source_rgb(cr, 0.6, 0.6, 0.65);
259 cairo_move_to(cr, 6, yy + 3);
260 cairo_show_text(cr, lbl);
261 }
262
263 /* X axis time gridlines/labels (relative seconds). */
264 int xdiv = 10;
265 for (int i = 0; i <= xdiv; i++) {
266 double xx = x0 + (x1 - x0) * i / (double)xdiv;
267 cairo_set_source_rgb(cr, 0.22, 0.22, 0.25);
268 cairo_move_to(cr, xx, y0);
269 cairo_line_to(cr, xx, y1);
270 cairo_stroke(cr);
271 char lbl[16];
272 double rel = -winw + winw * i / (double)xdiv;
273 snprintf(lbl, sizeof(lbl), "%.0fs", rel);
274 cairo_set_source_rgb(cr, 0.6, 0.6, 0.65);
275 cairo_move_to(cr, xx - 8, y1 + 16);
276 cairo_show_text(cr, lbl);
277 }
278
279 /* Plot frame. */
280 cairo_set_source_rgb(cr, 0.4, 0.4, 0.45);
281 cairo_rectangle(cr, x0, y0, x1 - x0, y1 - y0);
282 cairo_stroke(cr);
283
284 if (!s_plot.series)
285 return FALSE;
286
287 int enabled = 0;
288 for (int i = 0; i < s_plot.n_series; i++)
289 if (s_plot.series[i].enabled)
290 enabled++;
291
292 if (enabled == 0) {
293 cairo_set_source_rgb(cr, 0.7, 0.7, 0.75);
294 cairo_set_font_size(cr, 13);
295 cairo_move_to(cr, x0 + 14, y0 + 24);
296 cairo_show_text(cr,
297 "Select one or more signals on the left to plot them over time.");
298 return FALSE;
299 }
300
301 /* Traces. */
302 for (int i = 0; i < s_plot.n_series; i++) {
303 plot_series_t *ps = &s_plot.series[i];
304 if (!ps->enabled || ps->count == 0)
305 continue;
306
307 double span = ps->dmax - ps->dmin;
308 double base = ps->dmin;
309 if (!(span > 0.0)) {
310 /* Invalid/empty DBC range — auto-scale from visible samples. */
311 double mn = 1e300, mx = -1e300;
312 for (int k = 0; k < ps->count; k++) {
313 int idx = (ps->head - ps->count + k + PLOT_RING) % PLOT_RING;
314 if (ps->t[idx] < tmin) continue;
315 if (ps->v[idx] < mn) mn = ps->v[idx];
316 if (ps->v[idx] > mx) mx = ps->v[idx];
317 }
318 if (mx > mn) { base = mn; span = mx - mn; }
319 else { base = mn - 0.5; span = 1.0; }
320 }
321
322 /* Stride-decimate so we draw at most ~2 segments per horizontal pixel. */
323 int visible = 0;
324 int first = -1;
325 for (int k = 0; k < ps->count; k++) {
326 int idx = (ps->head - ps->count + k + PLOT_RING) % PLOT_RING;
327 if (ps->t[idx] >= tmin) { if (first < 0) first = k; visible++; }
328 }
329 if (visible == 0)
330 continue;
331 int max_pts = (int)((x1 - x0) * 2.0);
332 int stride = visible > max_pts ? visible / max_pts : 1;
333
334 cairo_set_source_rgb(cr, ps->col[0], ps->col[1], ps->col[2]);
335 cairo_set_line_width(cr, 1.5);
336 gboolean started = FALSE;
337 for (int k = first; k < ps->count; k += stride) {
338 int idx = (ps->head - ps->count + k + PLOT_RING) % PLOT_RING;
339 double tt = ps->t[idx];
340 if (tt < tmin) continue;
341 double x = x0 + (tt - tmin) / winw * (x1 - x0);
342 double norm = (ps->v[idx] - base) / span;
343 double y = norm_to_y(norm, y0, y1);
344 if (!started) { cairo_move_to(cr, x, y); started = TRUE; }
345 else cairo_line_to(cr, x, y);
346 }
347 cairo_stroke(cr);
348 }
349
350 /* Legend (top-left, over the plot). */
351 cairo_set_font_size(cr, 11);
352 double ly = y0 + 14;
353 for (int i = 0; i < s_plot.n_series; i++) {
354 plot_series_t *ps = &s_plot.series[i];
355 if (!ps->enabled)
356 continue;
357 cairo_set_source_rgb(cr, ps->col[0], ps->col[1], ps->col[2]);
358 cairo_rectangle(cr, x0 + 8, ly - 8, 12, 10);
359 cairo_fill(cr);
360 char txt[2 * DBC_NAME_MAX + 64];
361 if (ps->has_last)
362 snprintf(txt, sizeof(txt), "%s = %.3g %s",
363 ps->label, ps->last, ps->unit);
364 else
365 snprintf(txt, sizeof(txt), "%s", ps->label);
366 cairo_move_to(cr, x0 + 26, ly + 1);
367 cairo_show_text(cr, txt);
368 ly += 15;
369 if (ly > y1 - 4)
370 break;
371 }
372 return FALSE;
373}

378static gboolean plot_tick(gpointer data)
379{
380 (void)data;
381 if (s_plot.area && gtk_widget_get_mapped(s_plot.area)) {
382 /* Repaint continuously so the time axis scrolls smoothly; the data is
383 * cheap to redraw and we only run while the page is on screen. */
384 gtk_widget_queue_draw(s_plot.area);
385 s_plot.dirty = FALSE;
386 }
387 return G_SOURCE_CONTINUE;
