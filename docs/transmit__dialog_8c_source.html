68static int parse_hex_u32(const char *s, uint32_t max, uint32_t *out)
69{
70 if (!s || !out) return -1;
71
72 while (isspace((unsigned char)*s)) s++;
73 if (*s == '\0' || *s == '+' || *s == '-') return -1;
74
75 errno = 0;
76 char *end = NULL;
77 unsigned long v = strtoul(s, &end, 16);
78 if (errno == ERANGE || end == s || v > max) return -1;
79
80 while (isspace((unsigned char)*end)) end++;
81 if (*end != '\0') return -1;
82
83 *out = (uint32_t)v;
84 return 0;
85}

92static gboolean build_frame(can_msg_t *msg)
93{
94 memset(msg, 0, sizeof(*msg));
95
96 msg->is_extended = gtk_toggle_button_get_active(
97 GTK_TOGGLE_BUTTON(s_tx.ext_check));
98 msg->is_remote = gtk_toggle_button_get_active(
99 GTK_TOGGLE_BUTTON(s_tx.rtr_check));
100 msg->is_fd = gtk_toggle_button_get_active(
101 GTK_TOGGLE_BUTTON(s_tx.fd_check));
102 msg->is_brs = msg->is_fd && gtk_toggle_button_get_active(
103 GTK_TOGGLE_BUTTON(s_tx.brs_check));
104
105 if (msg->is_fd && msg->is_remote) {
106 gtk_label_set_markup(GTK_LABEL(s_tx.status_label),
107 "<span color='red'>CAN FD does not support RTR frames</span>");
108 return FALSE;
109 }
110
111 if (msg->is_fd && !g_app.fd_mode) {
112 gtk_label_set_markup(GTK_LABEL(s_tx.status_label),
113 "<span color='red'>Connection is not in CAN FD mode</span>");
114 return FALSE;
115 }
116
117 const char *id_str = gtk_entry_get_text(GTK_ENTRY(s_tx.id_entry));
118 uint32_t id_max = msg->is_extended ? TX_EXT_ID_MAX : TX_STD_ID_MAX;
119 if (parse_hex_u32(id_str, id_max, &msg->id) != 0) {
120 gtk_label_set_markup(GTK_LABEL(s_tx.status_label),
121 msg->is_extended
122 ? "<span color='red'>Invalid 29-bit ID</span>"
123 : "<span color='red'>Invalid 11-bit ID</span>");
124 return FALSE;
125 }
126
127 msg->dlc = (uint8_t)gtk_spin_button_get_value_as_int(
128 GTK_SPIN_BUTTON(s_tx.dlc_spin));
129
130 uint8_t max_dlc = msg->is_fd ? 64 : 8;
131 if (msg->dlc > max_dlc) msg->dlc = max_dlc;
132
133 if (!msg->is_remote) {
134 for (uint8_t i = 0; i < msg->dlc; i++) {
135 const char *ds =
136 gtk_entry_get_text(GTK_ENTRY(s_tx.data_entry[i]));
137 uint32_t v = 0;
138 if (strlen(ds) > 0 && parse_hex_u32(ds, TX_BYTE_MAX, &v) != 0) {
139 char buf[64];
140 snprintf(buf, sizeof(buf),
141 "<span color='red'>Invalid byte at pos %u</span>",
142 i);
143 gtk_label_set_markup(GTK_LABEL(s_tx.status_label), buf);
144 return FALSE;
145 }
146 msg->data[i] = (uint8_t)(v & 0xFF);
147 }
148 }
149 return TRUE;
150}

153static void send_frame(void)
154{
155 if (!g_app.connected) {
156 gtk_label_set_markup(GTK_LABEL(s_tx.status_label),
157 "<span color='red'>Not connected.</span>");
158 return;
159 }
160 can_msg_t msg;
161 if (!build_frame(&msg)) return;
162
163 can_msg_t *copy = malloc(sizeof(can_msg_t));
164 if (!copy) return;
165 *copy = msg;
166 g_async_queue_push(g_app.tx_queue, copy);
167
168 gtk_label_set_markup(GTK_LABEL(s_tx.status_label),
169 "<span color='green'>Queued.</span>");
170}

218static void update_data_fields(void)
219{
220 int is_fd = gtk_toggle_button_get_active(

222 int is_rtr = gtk_toggle_button_get_active(

231 dlc = gtk_spin_button_get_value_as_int(GTK_SPIN_BUTTON(s_tx.dlc_spin));

233 gtk_widget_set_sensitive(s_tx.brs_check, is_fd);
234
235 for (int i = 0; i < MAX_DATA_ENTRIES; i++) {
236 gboolean en = !is_rtr && (i < dlc) && (i < max_dlc);
237 gtk_widget_set_sensitive(s_tx.data_entry[i], en);
