44void gui_format_id(char *buf, size_t sz, uint32_t id, int is_ext)
45{
46 if (sz == 0) return;

49 snprintf(buf, sz, "%u", id);
50 } else if (is_ext) {

54 }

57void gui_format_data(char *buf, size_t sz, const uint8_t *d, uint8_t dlc)
58{
59 if (sz == 0) return;
60 buf[0] = '\0';
61
62 if (dlc == 0) {
63 snprintf(buf, sz, "-");
64 return;
65 }
66
67 size_t pos = 0;
68 if (g_app.data_format == APP_DATA_FORMAT_ASCII) {
69 for (uint8_t i = 0; i < dlc && pos + 1 < sz; i++) {
70 unsigned char c = d[i];
71 buf[pos++] = isprint(c) ? (char)c : '.';
72 }
73 buf[pos] = '\0';
74 return;
75 }
76
77 for (uint8_t i = 0; i < dlc && pos + 4 < sz; i++) {
78 if (g_app.data_format == APP_DATA_FORMAT_DEC) {
79 pos += (size_t)snprintf(buf + pos, sz - pos,
80 i ? " %u" : "%u", d[i]);
81 } else {
82 pos += (size_t)snprintf(buf + pos, sz - pos,
83 i ? " %02X" : "%02X", d[i]);
84 }
85 }
86}

105static int hex_value(char c)
106{
107 if (c >= '0' && c <= '9') return c - '0';
108 if (c >= 'a' && c <= 'f') return c - 'a' + 10;
109 if (c >= 'A' && c <= 'F') return c - 'A' + 10;
110 return -1;
111}

118static uint8_t parse_raw_data(const char *raw, uint8_t *out, uint8_t dlc)
119{
120 if (!raw || !out) return 0;
121
122 uint8_t count = 0;
123 while (count < dlc && raw[count * 2] && raw[count * 2 + 1]) {
124 int hi = hex_value(raw[count * 2]);
125 int lo = hex_value(raw[count * 2 + 1]);
126 if (hi < 0 || lo < 0)
127 break;
128 out[count] = (uint8_t)((hi << 4) | lo);
129 count++;
130 }
131 return count;
132}

135static gint64 timespec_to_ns(const struct timespec *ts)
136{
137 return ((gint64)ts->tv_sec * 1000000000LL) + (gint64)ts->tv_nsec;

141static void format_timestamp(char *buf, size_t sz, const struct timespec *ts)
142{
143 if (sz == 0) return;
144
145 time_t sec = ts->tv_sec;
146 struct tm tm_info;
147 if (!localtime_r(&sec, &tm_info)) {
148 snprintf(buf, sz, "--:--:--.---");
149 return;
150 }
151
152 snprintf(buf, sz, "%02d:%02d:%02d.%03ld",
153 tm_info.tm_hour, tm_info.tm_min, tm_info.tm_sec,
154 ts->tv_nsec / 1000000L);
155}

158static void format_interval(char *buf, size_t sz, gint64 delta_ns)
159{
160 if (sz == 0) return;
161 if (delta_ns <= 0) {
162 snprintf(buf, sz, "-");
163 return;
164 }
165
166 snprintf(buf, sz, "%.3f", (double)delta_ns / 1000000.0);
167}
168
170static void format_frequency(char *buf, size_t sz, gint64 delta_ns)
171{
172 if (sz == 0) return;
173 if (delta_ns <= 0) {
174 snprintf(buf, sz, "-");
175 return;
176 }
177
178 double hz = 1000000000.0 / (double)delta_ns;
179 if (hz >= 1000.0) {
180 snprintf(buf, sz, "%.0f", hz);
181 } else if (hz >= 10.0) {
182 snprintf(buf, sz, "%.1f", hz);
183 } else {
184 snprintf(buf, sz, "%.3f", hz);
185 }
186}

380/* Add message to trace (called from GTK main thread via idle_add) */
381/* ------------------------------------------------------------------ */
382
383void gui_add_message(const can_msg_t *msg)
384{
385 if (!g_gui.trace_store) return;
386
387 /* RX rows roll up by CAN ID; optional dedup mode applies the same to TX. */
388 if (should_rollup_message(msg)) {
389 GtkTreeIter iter;
390 gboolean found = FALSE;
391 if (gtk_tree_model_get_iter_first(
392 GTK_TREE_MODEL(g_gui.trace_store), &iter)) {
393 do {
394 guint row_id = 0;
395 guint row_dir = 0;
396 gboolean row_ext = FALSE;
397 gtk_tree_model_get(GTK_TREE_MODEL(g_gui.trace_store),
398 &iter,
399 TCOL_ID_RAW, &row_id,
400 TCOL_EXT_RAW, &row_ext,
401 TCOL_DIR_RAW, &row_dir,
402 -1);
403 if (row_id == msg->id &&
404 row_ext == (gboolean)msg->is_extended &&
405 row_dir == (guint)msg->direction) {
406 found = TRUE;
407
408 guint cnt = 0;
409 gint64 last_ts_ns = 0;
410 gtk_tree_model_get(GTK_TREE_MODEL(g_gui.trace_store),
411 &iter,
412 TCOL_COUNT, &cnt,
413 TCOL_TS_NS, &last_ts_ns,
414 -1);
415
416 gint64 ts_ns = timespec_to_ns(&msg->timestamp);
417 gint64 delta_ns = last_ts_ns > 0 ? ts_ns - last_ts_ns : 0;
418 if (delta_ns < 0)
419 delta_ns = 0;
420
421 char ts[TIME_TEXT_MAX];
422 char data_buf[DATA_TEXT_MAX];
423 char raw_data[RAW_DATA_MAX];
424 char interval_buf[INTERVAL_TEXT_MAX];
425 char freq_buf[FREQ_TEXT_MAX];
426 format_timestamp(ts, sizeof(ts), &msg->timestamp);
427 gui_format_data(data_buf, sizeof(data_buf),
428 msg->data, msg->dlc);
429 format_raw_data(raw_data, sizeof(raw_data),
430 msg->data, msg->dlc);
431 format_interval(interval_buf, sizeof(interval_buf),
432 delta_ns);
433 format_frequency(freq_buf, sizeof(freq_buf), delta_ns);
434
435 gtk_list_store_set(g_gui.trace_store, &iter,
436 TCOL_TIME, ts,
437 TCOL_TYPE, gui_msg_type_str(msg),
438 TCOL_DLC, (guint)msg->dlc,
439 TCOL_DATA, data_buf,
440 TCOL_INTERVAL, interval_buf,
441 TCOL_FREQ, freq_buf,
442 TCOL_COUNT, cnt + 1,
443 TCOL_FG, row_fg(msg),
444 TCOL_DATA_RAW, raw_data,
445 TCOL_TS_NS, ts_ns,
446 -1);
447 break;
448 }
449 } while (gtk_tree_model_iter_next(
450 GTK_TREE_MODEL(g_gui.trace_store), &iter));
451 }
452 if (found) return;
453 }
454
455 /* Enforce max rows only when a new visible row is appended. */
456 gint n = gtk_tree_model_iter_n_children(
457 GTK_TREE_MODEL(g_gui.trace_store), NULL);
458 if (n >= MAX_TRACE_ROWS) {
459 GtkTreeIter first;
460 if (gtk_tree_model_get_iter_first(
461 GTK_TREE_MODEL(g_gui.trace_store), &first))
462 gtk_list_store_remove(g_gui.trace_store, &first);
463 }
464
465 /* Format fields */
466 char id_buf[ID_TEXT_MAX];
467 char data_buf[DATA_TEXT_MAX];
468 char raw_data[RAW_DATA_MAX];
469 char ts[TIME_TEXT_MAX];
470 char interval_buf[INTERVAL_TEXT_MAX];
471 char freq_buf[FREQ_TEXT_MAX];
472 gint64 ts_ns = timespec_to_ns(&msg->timestamp);
473 gui_format_id(id_buf, sizeof(id_buf), msg->id, msg->is_extended);
474 gui_format_data(data_buf, sizeof(data_buf), msg->data, msg->dlc);
475 format_raw_data(raw_data, sizeof(raw_data), msg->data, msg->dlc);
476 format_timestamp(ts, sizeof(ts), &msg->timestamp);
477 format_interval(interval_buf, sizeof(interval_buf), 0);
478 format_frequency(freq_buf, sizeof(freq_buf), 0);
479
480 GtkTreeIter iter;
481 gtk_list_store_append(g_gui.trace_store, &iter);
482 gtk_list_store_set(g_gui.trace_store, &iter,
483 TCOL_SEQ, msg->seq,
484 TCOL_DIR, msg->direction == CAN_DIR_TX ? "Tx" : "Rx",
485 TCOL_TIME, ts,
486 TCOL_ID, id_buf,
487 TCOL_TYPE, gui_msg_type_str(msg),
488 TCOL_DLC, (guint)msg->dlc,
489 TCOL_DATA, data_buf,
490 TCOL_INTERVAL, interval_buf,
491 TCOL_FREQ, freq_buf,
492 TCOL_COUNT, 1u,
493 TCOL_FG, row_fg(msg),
494 TCOL_ID_RAW, (guint)msg->id,
495 TCOL_EXT_RAW, (gboolean)msg->is_extended,
496 TCOL_DIR_RAW, (guint)msg->direction,
497 TCOL_DATA_RAW, raw_data,
498 TCOL_TS_NS, ts_ns,
499 -1);
500
501 /* Auto-scroll to bottom */
502 GtkTreePath *path = gtk_tree_model_get_path(
503 GTK_TREE_MODEL(g_gui.trace_store), &iter);
504 gtk_tree_view_scroll_to_cell(GTK_TREE_VIEW(g_gui.trace_view),
505 path, NULL, FALSE, 0, 0);
506 gtk_tree_path_free(path);
507}

519void gui_refresh_trace_display(void)
520{
521 if (!g_gui.trace_store) return;
522
523 GtkTreeIter iter;
524 if (!gtk_tree_model_get_iter_first(
525 GTK_TREE_MODEL(g_gui.trace_store), &iter))
526 return;
527
528 do {
529 guint id = 0;
530 guint dlc = 0;
531 gboolean is_ext = FALSE;
532 gchar *raw = NULL;
533
534 gtk_tree_model_get(GTK_TREE_MODEL(g_gui.trace_store), &iter,
535 TCOL_ID_RAW, &id,
536 TCOL_EXT_RAW, &is_ext,
537 TCOL_DLC, &dlc,
538 TCOL_DATA_RAW, &raw,
539 -1);
540
541 uint8_t data[CANFD_DATA_MAX] = {0};
542 if (dlc > CANFD_DATA_MAX)
543 dlc = CANFD_DATA_MAX;
544 parse_raw_data(raw, data, (uint8_t)dlc);
545
546 char id_buf[ID_TEXT_MAX];
547 char data_buf[DATA_TEXT_MAX];
548 gui_format_id(id_buf, sizeof(id_buf), id, is_ext);
549 gui_format_data(data_buf, sizeof(data_buf), data, (uint8_t)dlc);
550
551 gtk_list_store_set(g_gui.trace_store, &iter,
552 TCOL_ID, id_buf,
553 TCOL_DATA, data_buf,
554 -1);
555
556 g_free(raw);
557 } while (gtk_tree_model_iter_next(
558 GTK_TREE_MODEL(g_gui.trace_store), &iter));
559}
