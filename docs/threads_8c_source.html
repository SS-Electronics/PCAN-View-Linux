138static void trace_record(const can_msg_t *msg)
139{
140 if (!g_app.tracing) return;
141
142 pthread_mutex_lock(&g_app.trace_mutex);
143 if (g_app.tracing && g_app.trace_len < TRACE_BUF_MAX) {
144 if (g_app.trace_len >= g_app.trace_cap) {
145 size_t ncap = g_app.trace_cap ? g_app.trace_cap * 2 : 4096;
146 can_msg_t *nb = realloc(g_app.trace_buf,
147 ncap * sizeof(can_msg_t));
148 if (nb) {
149 g_app.trace_buf = nb;
150 g_app.trace_cap = ncap;
151 }
152 }
153 if (g_app.trace_len < g_app.trace_cap)
154 g_app.trace_buf[g_app.trace_len++] = *msg;
155 }
156 pthread_mutex_unlock(&g_app.trace_mutex);
157}

163void *thread_rx(void *arg)
164{
165 (void)arg;
166
167 while (g_app.rx_running) {
168 can_msg_t msg;
169 int ret = drv_can_recv(&msg, 100);
170
171 if (ret != DRV_CAN_OK) continue;
172
173 msg.seq = __atomic_add_fetch(&g_app.msg_seq, 1,
174 __ATOMIC_SEQ_CST);
175 msg.direction = CAN_DIR_RX;
176
177 if (msg.is_error) {
178 __atomic_add_fetch(&g_app.error_count, 1, __ATOMIC_SEQ_CST);
179 } else {
180 __atomic_add_fetch(&g_app.rx_count, 1, __ATOMIC_SEQ_CST);
181 /* Accumulate bits for bus-load calculation */
182 uint64_t frame_bits = (uint64_t)(msg.is_extended ? 67u : 47u)
183 + (uint64_t)msg.dlc * 8u;
184 __atomic_add_fetch(&g_app.period_bits, frame_bits,
185 __ATOMIC_SEQ_CST);
186 }
187
188 /* Deliver to GUI */
189 can_msg_t *copy = malloc(sizeof(can_msg_t));
190 if (copy) {
191 *copy = msg;
192 gdk_threads_add_idle(idle_add_message, copy);
193 }
194
195 trace_record(&msg);
196 }
197 return NULL;
198}

204void *thread_tx(void *arg)
205{
206 (void)arg;
207
208 while (g_app.tx_running) {
209 /* Wait up to 100 ms for a message to appear. g_async_queue_timeout_pop()
210 * takes a RELATIVE timeout in microseconds (not an absolute time); using
211 * an absolute monotonic time here made the thread block for ~uptime and
212 * never observe tx_running == 0, hanging app_do_disconnect's join (which
213 * froze the app on close/disconnect). */
214 can_msg_t *msg = (can_msg_t *)
215 g_async_queue_timeout_pop(g_app.tx_queue,
216 100 * G_TIME_SPAN_MILLISECOND);
217
218 if (!msg) continue;
219
220 int ret = drv_can_send(msg);
221 if (ret == DRV_CAN_OK) {
222 msg->seq = __atomic_add_fetch(&g_app.msg_seq, 1,
223 __ATOMIC_SEQ_CST);
224 msg->direction = CAN_DIR_TX;
225 clock_gettime(CLOCK_REALTIME, &msg->timestamp);
226 __atomic_add_fetch(&g_app.tx_count, 1, __ATOMIC_SEQ_CST);
227
228 gdk_threads_add_idle(idle_update_stats, NULL);
229 trace_record(msg);
230 }
231 free(msg);
232 }
233 return NULL;
234}

239
240void *thread_stats(void *arg)
241{
242 (void)arg;
243
244 unsigned int ticks = 0;
245 while (g_app.stats_running) {
246 struct timespec ts = { 0, 50000000L }; /* 50 ms slices → responds within 50 ms */
247 nanosleep(&ts, NULL);
248
249 if (++ticks < 10) continue; /* accumulate 10 × 50 ms = 500 ms period */
250 ticks = 0;
251
252 if (!g_app.connected) continue;
253
254 /* Atomically collect bits accumulated in the last period */
255 uint64_t bits = __atomic_exchange_n(&g_app.period_bits, 0,
256 __ATOMIC_SEQ_CST);
257
258 /* bits / 0.5 s = bits/s; load% = bits_per_sec / bitrate * 100 */
259 double load = (double)(bits * 2) / (double)g_app.bitrate * 100.0;
260 if (load > 100.0) load = 100.0;
261
262 /* Double assignment is naturally atomic on x86-64 for aligned values */
263 g_app.bus_load = load;
264
265 gdk_threads_add_idle(idle_update_stats, NULL);
266 }
267 return NULL;
268}

274void app_do_connect(void)
275{
276 if (g_app.connected) {
277 gui_show_error(g_gui.window, "Error", "Already connected.");
278 return;
279 }
280
281 can_driver_t *drv = drv_can_get_socketcan();
282 int ret = drv_can_init(drv,
283 g_app.iface,
284 g_app.bitrate,
285 g_app.data_bitrate,
286 g_app.fd_mode,
287 g_app.listen_only,
288 &g_app.bit_timing);
289 if (ret != DRV_CAN_OK) {
290 char msg[256];
291 snprintf(msg, sizeof(msg),
292 "Failed to connect to %s:\n%s\n\n"
293 "Tip: ensure the interface exists and you have sufficient "
294 "privileges (or run with sudo).",
295 g_app.iface, drv_can_error_string(ret));
296 gui_show_error(g_gui.window, "Connection Error", msg);
297 return;
298 }
299
300 /* Reset counters */
301 __atomic_store_n(&g_app.rx_count, 0, __ATOMIC_SEQ_CST);
302 __atomic_store_n(&g_app.tx_count, 0, __ATOMIC_SEQ_CST);
303 __atomic_store_n(&g_app.error_count, 0, __ATOMIC_SEQ_CST);
304 __atomic_store_n(&g_app.msg_seq, 0, __ATOMIC_SEQ_CST);
305 __atomic_store_n(&g_app.period_bits, 0, __ATOMIC_SEQ_CST);
306 g_app.bus_load = 0.0;
307 g_app.bus_state = CAN_BUS_ACTIVE;
308 g_app.connected = 1;
309
310 /* Start background threads */
311 g_app.rx_running = 1;
312 g_app.tx_running = 1;
313 g_app.stats_running = 1;
314 pthread_create(&g_app.rx_thread, NULL, thread_rx, NULL);
315 pthread_create(&g_app.tx_thread, NULL, thread_tx, NULL);
316 pthread_create(&g_app.stats_thread, NULL, thread_stats, NULL);
317
318 /* Update toolbar sensitivity.
319 * NOTE: the Connect button stays enabled while connected so the user can
320 * re-open the connection dialog at any time (it prompts to disconnect
321 * first). Disabling it here was the cause of "the connection window is
322 * not enabled again" after a connect. */
323 if (g_gui.toolbar_connect)
324 gtk_widget_set_sensitive(g_gui.toolbar_connect, TRUE);
325 if (g_gui.toolbar_disconnect)
326 gtk_widget_set_sensitive(g_gui.toolbar_disconnect, TRUE);
327
328 /* Reflect the (possibly FD) link mode in the transmit panel so the per-row
329 * DLC can grow to 64 bytes when CAN FD is enabled. */
330 gui_tx_panel_update_fd();
331
332 gui_status_message("Connected to %s @ %u bps%s",
333 g_app.iface, g_app.bitrate,
334 g_app.listen_only ? " [listen-only]" : "");
335}

337void app_do_disconnect(void)
338{
339 if (!g_app.connected) return;
340
341 /* Signal idle callbacks to discard any late-arriving messages */
342 g_app.shutting_down = 1;
343
344 g_app.rx_running = 0;
345 g_app.tx_running = 0;
346 g_app.stats_running = 0;
347
348 pthread_join(g_app.rx_thread, NULL);
349 pthread_join(g_app.tx_thread, NULL);
350 pthread_join(g_app.stats_thread, NULL);
351
352 /* Drain any pending GTK idle callbacks while widgets still exist. */
353 while (g_main_context_iteration(NULL, FALSE)) {
354 /* keep flushing */
355 }
356
357 drv_can_deinit();
358 g_app.connected = 0;
359 g_app.shutting_down = 0;
360
361 /* Stop trace capture if running, but keep the captured frames so the user
362 * can still export them to CSV via Trace > Save after disconnecting. */
363 if (g_app.tracing)
364 g_app.tracing = 0;
365
366 /* Update toolbar sensitivity */
367 if (g_gui.toolbar_connect)
368 gtk_widget_set_sensitive(g_gui.toolbar_connect, TRUE);
369 if (g_gui.toolbar_disconnect)
370 gtk_widget_set_sensitive(g_gui.toolbar_disconnect, FALSE);
371 if (g_gui.toolbar_trace_start)
372 gtk_widget_set_sensitive(g_gui.toolbar_trace_start, FALSE);
373 if (g_gui.toolbar_trace_stop)
374 gtk_widget_set_sensitive(g_gui.toolbar_trace_stop, FALSE);
375
376 gui_update_stats();
377 gui_status_message("Disconnected from %s", g_app.iface);
378}
