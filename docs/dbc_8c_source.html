58static dbc_message_t *parse_bo(const char *line, dbc_db_t *db)
59{
60 /* BO_ <id> <name>: <dlc> <transmitter> */
61 unsigned long raw_id = 0;
62 char name[DBC_NAME_MAX] = {0};
63 unsigned int dlc = 0;
64
65 if (sscanf(line, "BO_ %lu %63[^:]: %u", &raw_id, name, &dlc) < 2)
66 return NULL;
67
68 /* Trim trailing whitespace from the captured name. */
69 size_t n = strlen(name);
70 while (n > 0 && isspace((unsigned char)name[n - 1]))
71 name[--n] = '\0';
72
73 dbc_message_t *m = dbc_push_message(db);
74 if (!m)
75 return NULL;
76
77 m->is_extended = (raw_id & DBC_EXT_FLAG) ? 1 : 0;
78 m->id = (uint32_t)(raw_id & ~DBC_EXT_FLAG);
79 m->dlc = (uint8_t)(dlc > 64 ? 64 : dlc);
80 snprintf(m->name, sizeof(m->name), "%s", name);
81 return m;
82}

90static int parse_sg(const char *line, dbc_message_t *m)
91{
92 if (!m || m->signal_count >= DBC_MAX_SIGNALS_PER_MSG)
93 return -1;
94
95 /* SG_ <name> : <start>|<len>@<order><sign> (<factor>,<offset>) [<min>|<max>] "<unit>" <recv> */
96 char name[DBC_NAME_MAX] = {0};
97 unsigned start = 0, len = 0;
98 int order = 1; /* 1 = Intel/LE, 0 = Motorola/BE */
99 char sign = '+';
100 double factor = 1.0, offset = 0.0;
101 double minv = 0.0, maxv = 0.0;
102 char unit[DBC_UNIT_MAX] = {0};
103
104 int got = sscanf(line,
105 " SG_ %63s : %u | %u @ %d %c ( %lf , %lf ) [ %lf | %lf ] \"%23[^\"]\"",
106 name, &start, &len, &order, &sign, &factor, &offset,
107 &minv, &maxv, unit);
108
109 /* Require at least up to the byte order; the rest may be absent in terse
110 * databases. (Multiplexed `SG_ name m0 :` forms are skipped — `name`
111 * would capture the multiplex token and the field count falls short.) */
112 if (got < 5)
113 return -1;
114
115 if (len == 0 || len > 64)
116 return -1;
117
118 dbc_signal_t *s = &m->signals[m->signal_count];
119 memset(s, 0, sizeof(*s));
120 snprintf(s->name, sizeof(s->name), "%s", name);
121 s->start_bit = (uint16_t)start;
122 s->length = (uint16_t)len;
123 s->little_endian = order ? 1 : 0;
124 s->is_signed = (sign == '-') ? 1 : 0;
125 s->factor = factor;
126 s->offset = offset;
127 s->min = minv;
128 s->max = maxv;
129 snprintf(s->unit, sizeof(s->unit), "%s", unit);
130
131 m->signal_count++;
132 return 0;
133}

135dbc_db_t *dbc_load_file(const char *path, char *err, size_t errsz)
136{
137 FILE *fp = fopen(path, "r");
138 if (!fp) {
139 if (err) snprintf(err, errsz, "Cannot open '%s'.", path);
140 return NULL;
141 }
142
143 dbc_db_t *db = calloc(1, sizeof(*db));
144 if (!db) {
145 fclose(fp);
146 if (err) snprintf(err, errsz, "Out of memory.");
147 return NULL;
148 }
149 strncpy(db->path, path, sizeof(db->path) - 1);
150
151 char line[1024];
152 dbc_message_t *cur = NULL;
153 while (fgets(line, sizeof(line), fp)) {
154 const char *p = line;
155 while (*p && isspace((unsigned char)*p))
156 p++;
157
158 if (strncmp(p, "BO_ ", 4) == 0) {
159 cur = parse_bo(p, db);
160 } else if (strncmp(p, "SG_ ", 4) == 0) {
161 if (parse_sg(p, cur) == 0)
162 db->signal_count++;
163 } else if (*p == '\0' || strncmp(p, "BO_TX_BU_", 9) == 0) {
164 /* Blank line or transmitter list — keep the current message. */
165 } else if (!isspace((unsigned char)line[0])) {
166 /* A new top-level (non-indented) record ends the current BO_. */
167 cur = NULL;
168 }
169 }
170 fclose(fp);
171
172 if (db->message_count == 0) {
173 if (err) snprintf(err, errsz,
174 "No CAN messages (BO_) found in '%s'.", path);
175 dbc_free(db);
176 return NULL;
177 }
178 return db;
179}

202uint64_t dbc_extract_raw(const uint8_t *data, uint8_t dlc,
203 const dbc_signal_t *sig)
204{
205 if (!data || !sig || sig->length == 0 || sig->length > 64)
206 return 0;
207
208 uint64_t value = 0;
209 const uint16_t len = sig->length;
210 const uint16_t total = (uint16_t)dlc * 8u;
211
212 if (sig->little_endian) {
213 /* Intel: bit i of the value comes from bit (start_bit + i). */
214 for (uint16_t i = 0; i < len; i++) {
215 uint16_t bit = (uint16_t)(sig->start_bit + i);
216 if (bit >= total)
217 break;
218 uint8_t b = data[bit >> 3];
219 value |= (uint64_t)((b >> (bit & 7)) & 1u) << i;
220 }
221 } else {
222 /* Motorola: consume MSB-first along the sawtooth bit order. */
223 int bit = sig->start_bit;
224 for (uint16_t i = 0; i < len; i++) {
225 if (bit < 0 || bit >= total)
226 break;
227 uint8_t b = data[bit >> 3];
228 value |= (uint64_t)((b >> (bit & 7)) & 1u) << (len - 1 - i);
229 if ((bit & 7) == 0)
230 bit += 15; /* jump to the MSB of the next byte */
231 else
232 bit -= 1;
233 }
234 }
235 return value;
236}

238double dbc_decode_physical(const dbc_signal_t *sig, uint64_t raw,
239 int64_t *signed_raw_out)
240{
241 int64_t sraw = (int64_t)raw;
242
243 if (sig->is_signed && sig->length < 64) {
244 uint64_t sign_bit = (uint64_t)1 << (sig->length - 1);
245 if (raw & sign_bit) {
246 /* Sign-extend by setting all bits above the field width. */
247 uint64_t mask = ~(((uint64_t)1 << sig->length) - 1);
248 sraw = (int64_t)(raw | mask);
249 }
250 }
251
252 if (signed_raw_out)
253 *signed_raw_out = sig->is_signed ? sraw : (int64_t)raw;
254
255 double base = sig->is_signed ? (double)sraw : (double)raw;
256 return base * sig->factor + sig->offset;
257}
