49static int socketcan_iface_name_valid(const char *iface)
50{
51 if (!iface || iface[0] == '\0')
52 return 0;
53
54 size_t len = strlen(iface);
55 if (len >= IFNAMSIZ || len >= SOCKETCAN_MAX_IFACE)
56 return 0;
57
58 for (size_t i = 0; i < len; i++) {
59 unsigned char c = (unsigned char)iface[i];
60 if (!isalnum(c) && c != '_' && c != '-' && c != '.')
61 return 0;
62 }
63 return 1;
64}

67static int iface_is_up(const char *iface)
68{
69 char path[300];
70 snprintf(path, sizeof(path), "/sys/class/net/%s/flags", iface);
71 FILE *f = fopen(path, "r");
72 if (!f)
73 return 0;
74 unsigned int flags = 0;
75 int n = fscanf(f, "%x", &flags);
76 fclose(f);
77 return (n == 1) && (flags & 1u /* IFF_UP */);
78}

98static int run_argv(const char *const argv[], int quiet_stderr)
99{
100 pid_t pid = fork();
101 if (pid < 0)
102 return -1;
103
104 if (pid == 0) {
105 if (quiet_stderr) {
106 int devnull = open("/dev/null", O_WRONLY);
107 if (devnull >= 0) {
108 dup2(devnull, STDERR_FILENO);
109 close(devnull);
110 }
111 }
112
113 execvp(argv[0], (char *const *)argv);
114 _exit(127);
115 }
116
117 int status = 0;
118 while (waitpid(pid, &status, 0) < 0) {
119 if (errno != EINTR)
120 return -1;
121 }
122
123 if (WIFEXITED(status) && WEXITSTATUS(status) == 0)
124 return 0;
125 return -1;
126}

138static int run_ip_priv(const char *const ip_argv[], int quiet_stderr)
139{
140 /* Fast path: succeeds when already running with sufficient privileges. */
141 if (run_argv(ip_argv, 1) == 0)
142 return 0;
143
144 /* Elevated path: pkexec <abs ip> <args…> (skip the leading "ip"). */
145 const char *pk_argv[40];
146 size_t cap = sizeof(pk_argv) / sizeof(pk_argv[0]);
147 size_t n = 0;
148 pk_argv[n++] = "pkexec";
149 pk_argv[n++] = ip_tool_path();
150 for (size_t i = 1; ip_argv[i] != NULL && n < cap - 1; i++)
151 pk_argv[n++] = ip_argv[i];
152 pk_argv[n] = NULL;
153
154 return run_argv(pk_argv, quiet_stderr);
155}

293int socketcan_open(socketcan_ctx_t *ctx, const char *iface,
294 uint32_t bitrate, uint32_t data_bitrate,
295 int fd_mode, int listen_only,
296 const can_bit_timing_t *timing)
297{
298 if (!ctx || !socketcan_iface_name_valid(iface))
299 return DRV_CAN_ERR_PARAM;
300
301 memset(ctx, 0, sizeof(*ctx));
302 ctx->sock_fd = -1;
303 ctx->fd_enabled = fd_mode;
304 ctx->listen_only = listen_only;
305 ctx->bitrate = bitrate;
306 ctx->data_bitrate = data_bitrate;
307 if (timing) {
308 ctx->timing = *timing;
309 } else {
310 ctx->timing.enabled = 0;
311 }
312 memcpy(ctx->iface, iface, strlen(iface) + 1);
313
314 /* Configure interface bitrate (skip initial vcan bitrate config) */
315 if (strncmp(iface, "vcan", 4) != 0) {
316 if (configure_interface(iface, bitrate, data_bitrate,
317 fd_mode, listen_only, timing) != 0)
318 return DRV_CAN_ERR_INIT;
319 } else {
320 /* For vcan interfaces, reuse an existing one when available; otherwise
321 * create it. Creation/bring-up needs CAP_NET_ADMIN, so run_ip_priv
322 * elevates through pkexec (graphical auth – no terminal sudo). */
323 if (if_nametoindex(iface) == 0) {
324 const char *add_args[] = {
325 "ip", "link", "add", "dev", iface,
326 "type", "vcan", NULL
327 };
328 if (run_ip_priv(add_args, 0) != 0 || if_nametoindex(iface) == 0) {
329 fprintf(stderr, "socketcan: failed to create %s\n", iface);
330 return DRV_CAN_ERR_INIT;
331 }
332 }
333
334 /* Ensure the interface is up (a freshly created vcan starts down, and
335 * a down interface neither transmits nor receives). Skip when already
336 * up so we don't trigger an unnecessary pkexec authentication. */
337 if (!iface_is_up(iface)) {
338 const char *up_args[] = {
339 "ip", "link", "set", iface, "up", NULL
340 };
341 run_ip_priv(up_args, 1);
342 }
343 }
344
345 /* Create raw CAN socket */
346 ctx->sock_fd = socket(PF_CAN, SOCK_RAW, CAN_RAW);
347 if (ctx->sock_fd < 0) {
348 perror("socketcan: socket");
349 return DRV_CAN_ERR_INIT;
350 }
351
352 /* Enable error frames */
353 can_err_mask_t err_mask = CAN_ERR_MASK;
354 setsockopt(ctx->sock_fd, SOL_CAN_RAW, CAN_RAW_ERR_FILTER,
355 &err_mask, sizeof(err_mask));
356
357 /* Keep locally transmitted frames out of the receive path. */
358 int recv_own_msgs = 0;
359 setsockopt(ctx->sock_fd, SOL_CAN_RAW, CAN_RAW_RECV_OWN_MSGS,
360 &recv_own_msgs, sizeof(recv_own_msgs));
361
362 /* Enable CAN FD if requested */
363 if (fd_mode) {
364 int enable = 1;
365 if (setsockopt(ctx->sock_fd, SOL_CAN_RAW, CAN_RAW_FD_FRAMES,
366 &enable, sizeof(enable)) < 0) {
367 fprintf(stderr, "socketcan: CAN FD not supported on this "
368 "interface, falling back to classic CAN\n");
369 ctx->fd_enabled = 0;
370 }
371 }
372
373 /* Bind to the interface */
374 struct ifreq ifr;
375 memset(&ifr, 0, sizeof(ifr));
376 memcpy(ifr.ifr_name, iface, strlen(iface) + 1);
377 if (ioctl(ctx->sock_fd, SIOCGIFINDEX, &ifr) < 0) {
378 perror("socketcan: ioctl SIOCGIFINDEX");
379 close(ctx->sock_fd);
380 ctx->sock_fd = -1;
381 return DRV_CAN_ERR_INIT;
382 }
383
384 struct sockaddr_can addr;
385 memset(&addr, 0, sizeof(addr));
386 addr.can_family = AF_CAN;
387 addr.can_ifindex = ifr.ifr_ifindex;
388
389 if (bind(ctx->sock_fd, (struct sockaddr *)&addr, sizeof(addr)) < 0) {
390 perror("socketcan: bind");
391 close(ctx->sock_fd);
392 ctx->sock_fd = -1;
393 return DRV_CAN_ERR_INIT;
394 }
395
396 return DRV_CAN_OK;
397}

418int socketcan_send(socketcan_ctx_t *ctx, const can_msg_t *msg)
419{
420 if (!ctx || ctx->sock_fd < 0 || !msg)
421 return DRV_CAN_ERR_PARAM;
422
423 if (msg->is_fd && ctx->fd_enabled) {
424 struct canfd_frame frame;
425 memset(&frame, 0, sizeof(frame));
426 frame.can_id = msg->id;
427 if (msg->is_extended) frame.can_id |= CAN_EFF_FLAG;
428 if (msg->is_remote) frame.can_id |= CAN_RTR_FLAG;
429 frame.len = msg->dlc > CANFD_DATA_MAX ? CANFD_DATA_MAX : msg->dlc;
430 if (msg->is_brs) frame.flags |= CANFD_BRS;
431 if (msg->is_esi) frame.flags |= CANFD_ESI;
432 memcpy(frame.data, msg->data, frame.len);
433
434 ssize_t n = write(ctx->sock_fd, &frame, sizeof(frame));
435 if (n != sizeof(frame)) return DRV_CAN_ERR;
436 } else {
437 struct can_frame frame;
438 memset(&frame, 0, sizeof(frame));
439 frame.can_id = msg->id;
440 if (msg->is_extended) frame.can_id |= CAN_EFF_FLAG;
441 if (msg->is_remote) frame.can_id |= CAN_RTR_FLAG;
442 frame.can_dlc = msg->dlc > CAN_MAX_DLC ? CAN_MAX_DLC : msg->dlc;
443 memcpy(frame.data, msg->data, frame.can_dlc);
444
445 ssize_t n = write(ctx->sock_fd, &frame, sizeof(frame));
446 if (n != sizeof(frame)) return DRV_CAN_ERR;
447 }
448 return DRV_CAN_OK;
449}

451int socketcan_recv(socketcan_ctx_t *ctx, can_msg_t *msg, int timeout_ms)
452{
453 if (!ctx || ctx->sock_fd < 0 || !msg)
454 return DRV_CAN_ERR_PARAM;
455
456 /* Wait with timeout */
457 fd_set rfds;
458 FD_ZERO(&rfds);
459 FD_SET(ctx->sock_fd, &rfds);
460
461 struct timeval tv;
462 tv.tv_sec = timeout_ms / 1000;
463 tv.tv_usec = (timeout_ms % 1000) * 1000;
464
465 int ret = select(ctx->sock_fd + 1, &rfds, NULL, NULL,
466 timeout_ms >= 0 ? &tv : NULL);
467 if (ret < 0) return DRV_CAN_ERR;
468 if (ret == 0) return DRV_CAN_ERR_TIMEOUT;
469
470 memset(msg, 0, sizeof(*msg));
471 clock_gettime(CLOCK_REALTIME, &msg->timestamp);
472
473 /* Try CAN FD frame first */
474 struct canfd_frame fdf;
475 ssize_t n = read(ctx->sock_fd, &fdf, sizeof(fdf));
476 if (n < 0) return DRV_CAN_ERR;
477
478 if (n == (ssize_t)sizeof(struct canfd_frame) && ctx->fd_enabled) {
479 /* CAN FD frame */
480 msg->is_fd = 1;
481 msg->is_brs = (fdf.flags & CANFD_BRS) ? 1 : 0;
482 msg->is_esi = (fdf.flags & CANFD_ESI) ? 1 : 0;
483 msg->is_extended = (fdf.can_id & CAN_EFF_FLAG) ? 1 : 0;
484 msg->is_remote = (fdf.can_id & CAN_RTR_FLAG) ? 1 : 0;
485 msg->is_error = (fdf.can_id & CAN_ERR_FLAG) ? 1 : 0;
486 msg->id = fdf.can_id & (msg->is_extended ? CAN_EFF_MASK
487 : CAN_SFF_MASK);
488 msg->dlc = fdf.len;
489 memcpy(msg->data, fdf.data, fdf.len);
490 } else {
491 /* Classic CAN frame */
492 struct can_frame *cf = (struct can_frame *)&fdf;
493 msg->is_fd = 0;
494 msg->is_extended = (cf->can_id & CAN_EFF_FLAG) ? 1 : 0;
495 msg->is_remote = (cf->can_id & CAN_RTR_FLAG) ? 1 : 0;
496 msg->is_error = (cf->can_id & CAN_ERR_FLAG) ? 1 : 0;
497 msg->id = cf->can_id & (msg->is_extended ? CAN_EFF_MASK
498 : CAN_SFF_MASK);
499 msg->dlc = cf->can_dlc;
500 memcpy(msg->data, cf->data, cf->can_dlc);
501 }
502 msg->direction = CAN_DIR_RX;
503 return DRV_CAN_OK;
504}

574int socketcan_list_interfaces(char ifnames[][SOCKETCAN_MAX_IFACE],
575 int max_count)
576{
577 int count = 0;
578 DIR *d = opendir("/sys/class/net");
579 if (!d) return 0;
580
581 struct dirent *ent;
582 while ((ent = readdir(d)) != NULL && count < max_count) {
583 if (ent->d_name[0] == '.') continue;
584
585 char path[300];
586 snprintf(path, sizeof(path),
587 "/sys/class/net/%s/type", ent->d_name);
588 FILE *f = fopen(path, "r");
589 if (!f) continue;
590
591 int type = 0;
592 int n = fscanf(f, "%d", &type);
593 fclose(f);
594
595 if (n == 1 && type == 280 /* ARPHRD_CAN */) {
596 size_t nl = strlen(ent->d_name);
597 if (nl >= SOCKETCAN_MAX_IFACE) continue; /* name too long, skip */
598 memcpy(ifnames[count], ent->d_name, nl + 1);
599 count++;
600 }
601 }
602 closedir(d);
603 return count;
604}
