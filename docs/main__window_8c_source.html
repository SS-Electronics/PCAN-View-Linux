65static const char *find_asset_path(const char *name)
66{
67 static char buf[1024];
68
69 const char *env = getenv("PCAN_VIEW_ASSET_DIR");
70 if (env && *env) {
71 snprintf(buf, sizeof(buf), "%s/%s", env, name);
72 if (access(buf, R_OK) == 0) return buf;
73 }
74
75 char exe[512];
76 ssize_t n = readlink("/proc/self/exe", exe, sizeof(exe) - 1);
77 if (n > 0) {
78 exe[n] = '\0';
79 char *slash = strrchr(exe, '/');
80 if (slash) {
81 *slash = '\0';
82 const char *rel[] = {
83 "assets", "../assets", "../share/pcan-view", NULL
84 };
85 for (int i = 0; rel[i]; i++) {
86 snprintf(buf, sizeof(buf), "%s/%s/%s", exe, rel[i], name);
87 if (access(buf, R_OK) == 0) return buf;
88 }
89 }
90 }
91
92 const char *fixed[] = {
93 "assets",
94 "/usr/local/share/pcan-view",
95 "/usr/share/pcan-view",
96 NULL
97 };
98 for (int i = 0; fixed[i]; i++) {
99 snprintf(buf, sizeof(buf), "%s/%s", fixed[i], name);
100 if (access(buf, R_OK) == 0) return buf;
101 }
102 return NULL;
103}

285static void trace_csv_write_row(FILE *fp, const can_msg_t *msg)
286{
287 char id_buf[16];
288 char data_buf[3 * CANFD_DATA_MAX + 4];
289
290 if (msg->is_extended)
291 snprintf(id_buf, sizeof(id_buf), "%08X", msg->id);
292 else
293 snprintf(id_buf, sizeof(id_buf), "%03X", msg->id);
294
295 if (msg->dlc == 0) {
296 snprintf(data_buf, sizeof(data_buf), "-");
297 } else {
298 size_t pos = 0;
299 for (uint8_t i = 0; i < msg->dlc && i < CANFD_DATA_MAX &&
300 pos + 4 < sizeof(data_buf); i++) {
301 pos += (size_t)snprintf(data_buf + pos, sizeof(data_buf) - pos,
302 i ? " %02X" : "%02X", msg->data[i]);
303 }
304 }
305
306 double ts = (double)msg->timestamp.tv_sec
307 + (double)msg->timestamp.tv_nsec / 1e9;
308
309 fprintf(fp, "%llu,%.6f,%s,%s,%s,%u,%s\n",
310 (unsigned long long)msg->seq,
311 ts,
312 msg->direction == CAN_DIR_TX ? "Tx" : "Rx",
313 id_buf,
314 gui_msg_type_str(msg),
315 msg->dlc,
316 data_buf);
317}

322static void on_trace_save(GtkWidget *w, gpointer d)
323{
324 (void)w; (void)d;
325
326 if (g_app.trace_len == 0) {
327 gui_show_error(g_gui.window, "Save Trace",
328 "No trace data captured yet.\n"
329 "Use Trace > Start, generate some traffic, then Stop.");
330 return;
331 }
332
333 GtkWidget *dlg = gtk_file_chooser_dialog_new(
334 "Save Trace as CSV",
335 GTK_WINDOW(g_gui.window),
336 GTK_FILE_CHOOSER_ACTION_SAVE,
337 "_Cancel", GTK_RESPONSE_CANCEL,
338 "_Save", GTK_RESPONSE_ACCEPT,
339 NULL);
340 gtk_file_chooser_set_do_overwrite_confirmation(
341 GTK_FILE_CHOOSER(dlg), TRUE);
342 gtk_file_chooser_set_current_name(GTK_FILE_CHOOSER(dlg), "trace.csv");
343
344 GtkFileFilter *ff = gtk_file_filter_new();
345 gtk_file_filter_add_pattern(ff, "*.csv");
346 gtk_file_filter_set_name(ff, "CSV trace (*.csv)");
347 gtk_file_chooser_add_filter(GTK_FILE_CHOOSER(dlg), ff);
348
349 if (gtk_dialog_run(GTK_DIALOG(dlg)) == GTK_RESPONSE_ACCEPT) {
350 char *fn = gtk_file_chooser_get_filename(GTK_FILE_CHOOSER(dlg));
351 strncpy(g_app.trace_filename, fn, APP_TRACE_FILE_LEN - 1);
352 g_app.trace_filename[APP_TRACE_FILE_LEN - 1] = '\0';
353 g_free(fn);
354
355 FILE *fp = fopen(g_app.trace_filename, "w");
356 if (fp) {
357 fprintf(fp, "seq,timestamp,direction,id,type,dlc,data\n");
358 pthread_mutex_lock(&g_app.trace_mutex);
359 size_t n = g_app.trace_len;
360 for (size_t i = 0; i < n; i++)
361 trace_csv_write_row(fp, &g_app.trace_buf[i]);
362 pthread_mutex_unlock(&g_app.trace_mutex);
363 fclose(fp);
364 gui_status_message("Saved %zu frame(s) to %s",
365 n, g_app.trace_filename);
366 } else {
367 gui_show_error(g_gui.window, "Trace Error",
368 "Could not open file for writing.");
369 }
370 }
371 gtk_widget_destroy(dlg);
372}

794static int parse_hex_u32_strict(const char *s, uint32_t max, uint32_t *out)
795{
796 if (!s || !out) return -1;
797
798 while (isspace((unsigned char)*s)) s++;
799 if (*s == '\0' || *s == '+' || *s == '-') return -1;
800
801 errno = 0;
802 char *end = NULL;
803 unsigned long v = strtoul(s, &end, 16);
804 if (errno == ERANGE || end == s || v > max) return -1;
805
806 while (isspace((unsigned char)*end)) end++;
807 if (*end != '\0') return -1;
808
809 *out = (uint32_t)v;
810 return 0;
811}

1045static uint8_t canfd_round_len(int n)
1046{
1047 static const uint8_t valid[] = {
1048 0, 1, 2, 3, 4, 5, 6, 7, 8, 12, 16, 20, 24, 32, 48, 64
1049 };
1050 for (size_t i = 0; i < sizeof(valid) / sizeof(valid[0]); i++)
1051 if (n <= valid[i])
1052 return valid[i];
1053 return CANFD_DATA_MAX;
1054}

1060static gboolean txp_build_frame(tx_panel_row_t *row, can_msg_t *msg)
1061{
1062 if (!row || !msg)
1063 return FALSE;
1064
1065 memset(msg, 0, sizeof(*msg));
1066
1067 msg->is_extended = gtk_toggle_button_get_active(
1068 GTK_TOGGLE_BUTTON(row->ext_check));
1069 msg->is_remote = gtk_toggle_button_get_active(
1070 GTK_TOGGLE_BUTTON(row->rtr_check));
1071
1072 /* Frames are sent as CAN FD when the link is in FD mode (BRS enabled). */
1073 msg->is_fd = g_app.fd_mode ? 1 : 0;
1074 msg->is_brs = g_app.fd_mode ? 1 : 0;
1075
1076 int req_len = gtk_spin_button_get_value_as_int(
1077 GTK_SPIN_BUTTON(row->dlc_spin));
1078 if (!msg->is_fd && req_len > CAN_MAX_DLC)
1079 req_len = CAN_MAX_DLC;
1080
1081 const char *ids = gtk_entry_get_text(GTK_ENTRY(row->id_entry));
1082 uint32_t id_max = msg->is_extended ? TX_EXT_ID_MAX : TX_STD_ID_MAX;
1083 if (parse_hex_u32_strict(ids, id_max, &msg->id) != 0) {
1084 txp_set_error(row, msg->is_extended
1085 ? "Invalid 29-bit ID"
1086 : "Invalid 11-bit ID");
1087 return FALSE;
1088 }
1089
1090 if (!msg->is_remote) {
1091 for (int i = 0; i < req_len && i < TX_PANEL_DATA_COLS; i++) {
1092 const char *ds = gtk_entry_get_text(
1093 GTK_ENTRY(row->data_entry[i]));
1094 uint32_t value = 0;
1095 if (parse_hex_u32_strict(ds, TX_BYTE_MAX, &value) != 0) {
1096 char err[64];
1097 snprintf(err, sizeof(err), "Invalid data byte [%d]", i);
1098 txp_set_error(row, err);
1099 return FALSE;
1100 }
1101 msg->data[i] = (uint8_t)value;
1102 }
1103 }
1104
1105 /* CAN FD payloads must be a valid length; pad up if needed (extra bytes
1106 * are already zero from the memset above). */
1107 msg->dlc = msg->is_fd ? canfd_round_len(req_len) : (uint8_t)req_len;
1108 return TRUE;
1109}

1115static gboolean txp_queue_frame(int row_index, gboolean cyclic)
1116{
1117 tx_panel_row_t *row = txp_row_from_data(GINT_TO_POINTER(row_index));
1118 if (!row)
1119 return FALSE;
1120
1121 if (!g_app.connected) {
1122 txp_set_error(row, "Not connected");
1123 return FALSE;

1139 txp_set_status(row, "green", "Queued");
