40static void populate_interfaces(GtkComboBoxText *combo)
41{
42#define MAX_IF 32
43 char names[MAX_IF][SOCKETCAN_MAX_IFACE];
44 int n = socketcan_list_interfaces(names, MAX_IF);
45
46 /* Always add vcan0 as the first / fallback option */
47 gtk_combo_box_text_append_text(combo, "vcan0");
48
49 for (int i = 0; i < n; i++) {
50 if (strcmp(names[i], "vcan0") != 0)
51 gtk_combo_box_text_append_text(combo, names[i]);
52 }
53
54 /* Pre-select the interface currently stored in g_app */
55 GtkTreeModel *model =
56 gtk_combo_box_get_model(GTK_COMBO_BOX(combo));
57 GtkTreeIter iter;
58 gboolean valid =
59 gtk_tree_model_get_iter_first(model, &iter);
60 int idx = 0, sel = 0;
61 while (valid) {
62 gchar *txt = NULL;
63 gtk_tree_model_get(model, &iter, 0, &txt, -1);
64 if (txt && strcmp(txt, g_app.iface) == 0) sel = idx;
65 g_free(txt);
66 valid = gtk_tree_model_iter_next(model, &iter);
67 idx++;
68 }
69 gtk_combo_box_set_active(GTK_COMBO_BOX(combo), sel);
70#undef MAX_IF
71}
